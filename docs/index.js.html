¶ node.log

Default nodeGame standard out, override to redirect

Default behavior is to output a text in the form: nodeGame: some text.

Logs entries are displayed only if their verbosity level is greater than node.verbosity

Params
txt string The text to output
level string number Optional. The verbosity level of this log. Defaults, level = 0
prefix string Optional. A text to display at the beginning of the log entry. Defaults prefix = 'nodeGame: ' 
node.log = function (txt, level, prefix) {
  if ('undefined' === typeof txt) return false;
  
  level   = level || 0;
  prefix  = ('undefined' === typeof prefix)   ? 'nodeGame: '
                        : prefix;
  if ('string' === typeof level) {
    level = node.verbosity_levels[level];
  }
  if (node.verbosity > level) {
    console.log(prefix + txt);
  }
};
