¶ Outgoing listeners

Outgoing listeners are fired when messages are sent

 
(function (node) {

  if (!node) {
    console.log('nodeGame not found. Cannot add outgoing listeners');
    return false;
  }
  
  var GameMsg = node.GameMsg,
    GameState = node.GameState;
  
  var action = node.action,
    target = node.target;
  
  var say = action.SAY + '.',
    set = action.SET + '.',
    get = action.GET + '.',
    OUT  = node.OUT;
