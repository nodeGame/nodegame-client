¶ node.connect

Establishes a connection with a nodeGame server

Params
uri string Optional. The uri to connect to.
cb function Optional. A callback to execute as soon as the connection is established.
socketOptions object Optional. A configuration object for the socket connect method. 
    NGC.prototype.connect = function(uri, cb, socketOptions) {
        if (cb) {
            if ('function' !== typeof cb) {
                throw new TypeError('node.connect: cb must be function or ' +
                                    'undefined');
            }
            this.once('SOCKET_CONNECT', function() {
                cb();
            });
        }
        this.socket.connect(uri, socketOptions);
    };
