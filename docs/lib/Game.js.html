  Game.prototype.next = function(times) {
    if (!times) return gameLoop.next(gameState);
    return gameLoop.jumpTo(gameState, Math.abs(times));
  };
  
  Game.prototype.previous = function (times) {
    if (!times) return gameLoop.previous(gameState);
    return gameLoop.jumpTo(gameState, -Math.abs(times));

  Game.prototype.jumpTo = function (jump) {
