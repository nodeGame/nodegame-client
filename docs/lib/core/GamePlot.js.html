¶ GamePlot.jump

Returns a distant stage in the stager

Works with negative delta only in simple mode. Uses GamePlot.previous and GamePlot.next for stepping. If a sequence end is reached, returns immediately.

Params
curStage GameStage The GameStage of reference
delta number The offset. Negative number for backward stepping.
Returns
GameStage string The GameStage describing the distant stage
See
GameStage
See
GamePlot.previous
See
GamePlot.next 
    GamePlot.prototype.jump = function(curStage, delta) {
        if (delta < 0) {
            while (delta < 0) {
                curStage = this.previous(curStage);
                delta++;

                if (!(curStage instanceof GameStage) || curStage.stage === 0) {
                    return curStage;
                }
            }
        }
        else {
            while (delta > 0) {
                curStage = this.next(curStage);
                delta--;

                if (!(curStage instanceof GameStage)) {
                    return curStage;
                }
            }
        }

        return curStage;
    };

            i = -1, len = this.stager.sequence.length;

object null The corresponding step object, or NULL if the step was not found 
    GamePlot.prototype.getStep = function(gameStage) {
        var seqObj, stepObj, stageObj;

        if (!this.stager) return null;
        gameStage = new GameStage(gameStage);
        if ('number' === typeof gameStage.step) {
            seqObj = this.getSequenceObject(gameStage);
            if (seqObj) {
                stepObj = this.stager.steps[seqObj.steps[gameStage.step - 1]];
            }
        }
        else {
            stepObj = this.stager.steps[gameStage.step];
        }
        return stepObj || null;
    };
